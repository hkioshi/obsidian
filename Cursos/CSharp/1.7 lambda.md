# Com func
`Func` é um delegado que representa um método que retorna um valor; 
Func\<tipoArg1?,...,tipoArgN?,tipoRetorno> nome = 
(nomeArg1?,...,nomeArgN?) => 
{
	Codigo
	return Retorno
}

# Com Action
`Action` é um delegado que representa um método que não retorna valor.
Func\<tipoArg1?,...,tipoArgN?> nome = 
(nomeArg1?,...,nomeArgN?) => 
{
	Codigo
}
